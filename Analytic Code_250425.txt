*Calculate incidence rate;
proc sql; create table crude2 as select "Crude" as type, exp, count(distinct indi_dscm_no) as n_patient, sum(fu_dt) as s_pytime, sum(out) as n_event, (calculated n_event/calculated s_pytime*1000) as ir_p1000  from crude1 group by 1, 2; quit;
proc sql; create table match2 as select "Match" as type, exp, count(distinct indi_dscm_no) as n_patient, sum(fu_dt) as s_pytime, sum(out) as n_event, (calculated n_event/calculated s_pytime*1000) as ir_p1000  from match1 group by 1, 2; quit;

*Calculate hazard ratio;
proc phreg data=crude1; class exp (ref="0"); model fu_dt*out(0)=exp/ risklimits; ods output ParameterEstimates=crude3; quit; data crude4; set crude3; exp=ClassVal0*1; run;
proc phreg data=match1; class exp (ref="0"); model fu_dt*out(0)=exp/ risklimits; ods output ParameterEstimates=match3; quit; data match4; set match3; exp=ClassVal0*1; run;

*Calculate 95% CI for incidence rate;
ods output lsmeans=crude5; proc genmod data=crude1; class  exp(ref="0"); model out=exp / dist=poisson offset=l_fu_dt link=log; lsmeans exp/ cl; quit; ods output clear;  
data crude6; set crude5; ir=put((exp(estimate))*1000,5.3); ir_p1000_lci=put((exp(lower))*1000,5.3); ir_p1000_uci=put((exp(upper))*1000,5.3); run;  
ods output lsmeans=match5; proc genmod data=match1; class  exp(ref="0"); model out=exp / dist=poisson offset=l_fu_dt link=log; lsmeans exp/ cl; quit; ods output clear;  
data match6; set match5; ir=put((exp(estimate))*1000,5.3); ir_p1000_lci=put((exp(lower))*1000,5.3); ir_p1000_uci=put((exp(upper))*1000,5.3); run;  

proc sql; create table crude as select distinct a.*, c.ir_p1000_lci, c.ir_p1000_uci, b.hazardratio, b.hrlowercl, b.hruppercl, b.probchisq, "Crude" as type from crude2 as a left join crude4 as b on a.exp=b.exp left join crude6 as c on a.exp=c.exp; quit; 
proc sql; create table match as select distinct a.*, c.ir_p1000_lci, c.ir_p1000_uci, b.hazardratio, b.hrlowercl, b.hruppercl, b.probchisq, "Match" as type from match2 as a left join match4 as b on a.exp=b.exp left join match6 as c on a.exp=c.exp; quit; 
data results_EXP&integer.; set crude match; EXP=&integer.; run; 

%end;

%macro km_curve(integer);

ods output survivalplot=survivalplot;
proc lifetest data=temp1 plots=s(atrisk(outside)=0 to 30 by 2) outs=km (drop=sdf_lcl sdf_ucl) notable; time fu*out(0); strata exp/ test=logrank; quit;
data s1.km_crude_exp&integer.; set survivalplot; surv1=(1-survival); surv2=(1-survival)*100; surv3=survival*100; if stratum=0 then ant="Non-AD"; else if stratum=1 then ant="AD"; run;
